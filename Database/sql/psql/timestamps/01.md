```sql

-- Dates and Timestamps

SELECT NOW();
SELECT NOW():DATE;
SELECT NOW()::TIME;

-- Adding and subtracting with dates
SELECT NOW() - INTERVAL '1 YEAR';
SELECT NOW() - INTERVAL '5 YEARS';
SELECT NOW() - INTERVAL '5 MONTHS';
SELECT NOW() - INTERVAL '5 DAYS';
SELECT NOW() + INTERVAL '5 DAYS';
SELECT NOW()::DATE + INTERVAL '5 YEARS';



SELECT NOW()::DATE + INTERVAL '100 YEARS';
SELECT (NOW() + INTERVAL '100 YEARS')::DATE;

-- EXTRACTING FIELDS
SELECT EXTRACT (YEAR FROM NOW());
SELECT EXTRACT (MONTH FROM NOW());
SELECT EXTRACT (DOW FROM NOW()); --DAY OF THE WEEK 0 TO 6
SELECT EXTRACT (CENTURY FROM NOW());



-- AGE FUNCTION
SELECT first_name,last_name,gender,country_of_birth,data_of_birth, AGE(NOW(),data_of_birth) AS age FROM person;

```
