```jsx
import React from 'react';
import { createStore } from 'redux';

// STORE
// - Globalize state
// - One big state
// - Stores all the data for entire app

// ACTION
// - Describe the what you WANT NOT DOING

// REDUCER
// - Check the what action did based on that it will change data in store

//--------------//
// COUNTER APP  //
// -------------//

// ACTION
let increment = () => {
  return {
    type: 'INCREMENT',
  };
};

let decrement = () => {
  return {
    type: 'DECREMENT',
  };
};

// REDUCER
// initialize the state value state =0
const counter = (state = 0, action) => {
  switch (action.type) {
    case 'INCREMENT':
      return state + 1;
    case 'DECREMENT':
      return state - 1;
  }
};

// STORE
let store = createStore(counter);

// display the it in the console
store.subscribe(() => console.log(store.getState()));

// DISPATCH
store.dispatch(increment());

function App() {
  return (
    <div className="App">
      <h1>Hello world</h1>
    </div>
  );
}

export default App;
```
